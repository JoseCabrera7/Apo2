import java.util.Scanner;

public class ejer_par_impar {
	public static void main(String[] args) {
	try (Scanner sc = new Scanner(System.in)) {
		int n=sc.nextInt();
		if (n==1) {
			System.out.print("0");
		}
		else if (n%2==0) {
			System.out.print(" Numero par");
		}
		
		else if (n/2!=0){
			System.out.print("Numero impar");
		}
	}
	}
}
